üöÄ Quick Start

1. Install

npm install bitcoin-inscription-viewer

2. Basic Usage

import { InscriptionViewer } from 'bitcoin-inscription-viewer'

function App() {
  const inscriptions = [
    '6fb976ab49dcec017f1e201e84395983204ae1a7c2abf7ced0a85d692e442799i0'
  ]

  return (
    <InscriptionViewer
      inscriptions={inscriptions}
      cardSize={300}
      gridCols={3}
      showHeaders={true}
    />
  )
}

3. With LaserEyes Wallet

import { LaserEyesProvider } from '@omnisat/lasereyes-react'
import { LaserEyesInscriptionGallery } from 'bitcoin-inscription-viewer'

function App() {
  return (
    <LaserEyesProvider>
      <LaserEyesInscriptionGallery
        columns={3}
        cardSize={200}
        enableModal={true}
      />
    </LaserEyesProvider>
  )
}
