Error Boundary Implementation

import React from 'react';
import { InscriptionViewer, ErrorBoundary } from 'bitcoin-inscription-viewer';

function App() {
  return (
    <ErrorBoundary
      fallback={({ error, retry }) => (
        <div className="error-container">
          <h3>Something went wrong</h3>
          <p>{error.message}</p>
          <button onClick={retry}>Try Again</button>
        </div>
      )}
    >
      <InscriptionViewer inscriptionId="abc123" />
    </ErrorBoundary>
  );
}

Retry Configuration

import { OrdinalsApiService } from 'bitcoin-inscription-viewer';

const apiService = new OrdinalsApiService({
  retryConfig: {
    maxRetries: 3,
    baseDelay: 1000,      // 1 second
    maxDelay: 30000,      // 30 seconds
    backoffFactor: 2,     // Exponential backoff
    retryOn: [408, 429, 500, 502, 503, 504]
  }
});

// Automatically retries failed requests
const inscription = await apiService.getInscription('id');

Custom Error Handler

const handleInscriptionError = (error, context) => {
  switch (error.type) {
    case 'NETWORK_ERROR':
      return {
        message: 'Network connection failed',
        action: 'retry',
        fallback: 'cached-content'
      };
    
    case 'CONTENT_ERROR':
      return {
        message: 'Content format not supported',
        action: 'fallback',
        fallback: 'placeholder-image'
      };
    
    case 'NOT_FOUND':
      return {
        message: 'Inscription not found',
        action: 'redirect',
        fallback: null
      };
  }
};

Fallback Strategies

const InscriptionWithFallback = ({ inscriptionId }) => {
  const [error, setError] = useState(null);
