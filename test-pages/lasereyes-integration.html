LaserEyes provides direct access to Bitcoin node functionality, allowing you to fetch inscription content without relying on external APIs.

# Install dependencies
npm install @omnisat/lasereyes bitcoin-inscription-viewer

# Import and use
import { LaserEyesProvider } from '@omnisat/lasereyes';
import { LaserEyesInscriptionGallery } from 'bitcoin-inscription-viewer';

function App() {
  return (
    <LaserEyesProvider>
      <LaserEyesInscriptionGallery
        columns={3}
        cardSize={250}
        enableModal={true}
      />
    </LaserEyesProvider>
  );
}

// Custom content fetcher using LaserEyes
import { useLaserEyes } from '@omnisat/lasereyes';

function CustomInscriptionViewer() {
  const { getInscriptionContent, isConnected } = useLaserEyes();
  
  const customFetcher = async (inscriptionId) => {
    if (!isConnected) throw new Error('Wallet not connected');
    return await getInscriptionContent(inscriptionId);
  };

  return (
    <EnhancedInscriptionViewer
      inscriptions={inscriptionIds}
      contentFetcher={customFetcher}
      cacheConfig={{
        enabled: true,
        maxSize: 500,
        ttl: 600000
      }}
      performanceOptions={{
        batchSize: 10,
        maxConcurrency: 5
      }}
    />
  );
}

üì± Different View Modes

The LaserEyesInscriptionGallery supports multiple display modes:

// Gallery view (default)
<LaserEyesInscriptionGallery
  columns={3}
  cardSize={250}
  showIndex={true}
  enableModal={true}
/>

// Compact view
<LaserEyesInscriptionGallery
  columns={6}
  cardSize={120}
  showIndex={false}
  enableModal={true}
/>

// Showcase view
<LaserEyesInscriptionGallery
  columns={2}
  cardSize={300}
  showIndex={true}
  enableModal={true}
  filter={(inscription) => inscription.number < 1000} // Show rare inscriptions
/>

üõ†Ô∏è Error Handling & Loading States

function WalletInscriptionViewer() {
  const { isConnected, isConnecting, connect } = useLaserEyes();
  
  if (isConnecting) {
    return <div>Connecting to wallet...</div>;
  }
  
  if (!isConnected) {
    return (
      <div>
        <button onClick={connect}>Connect LaserEyes Wallet</button>
      </div>
    );
  }
  
  return (
    <LaserEyesInscriptionGallery
      columns={3}
      cardSize={200}
      enableModal={true}
      onError={(error) => console.error('Inscription error:', error)}
      onInscriptionClick={(inscription) => {
        // Handle clicks - could navigate, show details, etc.
        console.log('Clicked inscription:', inscription);
      }}
    />
  );
}
