import React from 'react';
import { InscriptionGallery } from 'bitcoin-inscription-viewer';

function GalleryDemo() {
  const sampleInscriptions = [
    {
      id: 'inscription1',
      inscription_number: 1234567,
      content_type: 'image/jpeg',
      content_url: 'https://ordinals.com/content/abc123...',
      preview_url: 'https://ordinals.com/preview/abc123...',
      genesis_transaction_id: 'abc123...',
      genesis_address: 'bc1qxy2kgdygjrsqtzq2n0yrf2493p83kkfjhx0wlh',
      output_value: 10000,
      sat_ordinal: 1234567890,
      sat_rarity: 'common',
      content_length: 50000,
      genesis_fee: 1000,
      genesis_height: 800000,
      inscription_id: 'abc123...i0',
      location: 'abc123...:0:0',
      output: 'abc123...:0',
      value: 546,
      offset: 0,
      timestamp: '2023-01-01T00:00:00.000Z'
    },
    // ... more inscriptions
  ];

  return (
    <div className="w-full h-[600px]">
      <InscriptionGallery
        inscriptions={sampleInscriptions}
        itemSize={200}
        gap={16}
        enableVirtualScrolling={true}
        onInscriptionClick={(inscription) => {
          console.log('Clicked inscription:', inscription);
        }}
        className="w-full h-full"
        cardClassName="bg-white/10 border border-white/20 rounded-lg"
      />
    </div>
  );
}

Virtual Scrolling

Efficiently handles thousands of inscriptions with smooth scrolling performance.

Lazy Loading

Images load only when they come into view, optimizing bandwidth usage.

Responsive Design

Automatically adapts to different screen sizes with configurable grid layouts.

Customizable Cards

Apply custom styling to gallery cards with className props.

Click Handling

Built-in click handlers for opening modals or navigating to details.

Performance Optimized

Uses React.memo and useMemo for optimal rendering performance.

Gallery Props

Prop Type Default Description
inscriptions Inscription[] required Array of inscription objects to display
itemSize number 200 Size of each gallery item in pixels
gap number 16 Gap between gallery items
enableVirtualScrolling boolean true Enable virtual scrolling for performance
onInscriptionClick function undefined Callback when an inscription is clicked
className string undefined CSS class for the gallery container
cardClassName string undefined CSS class for individual cards
