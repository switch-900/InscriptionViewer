Performance Monitor

import { PerformanceMonitor } from 'bitcoin-inscription-viewer';

function App() {
  const monitor = new PerformanceMonitor();
  
  useEffect(() => {
    // Start monitoring
    monitor.startMonitoring();
    
    // Get metrics
    const metrics = monitor.getMetrics();
    console.log('Cache hits:', metrics.cacheHits);
    console.log('API calls:', metrics.apiCalls);
    console.log('Avg response time:', metrics.avgResponseTime);
    
    return () => monitor.stopMonitoring();
  }, []);
}

Complete Optimization Setup

import React from 'react';
import {
  InscriptionViewer,
  OrdinalsApiService,
  InscriptionContentCache,
  PerformanceMonitor
} from 'bitcoin-inscription-viewer';

function OptimizedApp() {
  // Configure cache
  const cache = new InscriptionContentCache({
    maxSize: 1000,
    maxMemoryMB: 100,
    ttlMinutes: 30,
    enableCompression: true
  });

  // Configure API service with optimizations
  const apiService = new OrdinalsApiService({
    enableBatchFetching: true,
    batchSize: 50,
    batchDelay: 100,
    cache: cache
  });

  // Setup performance monitoring
  const monitor = new PerformanceMonitor();

  useEffect(() => {
    monitor.startMonitoring();
    
    // Register service worker
    if ('serviceWorker' in navigator) {
      navigator.serviceWorker.register('/inscription-sw.js');
    }

    return () => monitor.stopMonitoring();
  }, []);

  return (
    <div className="app">
      <InscriptionViewer
        inscriptionId="your-inscription-id"
        apiService={apiService}
        cache={cache}
        enableVirtualScrolling={true}
        enableLazyLoading={true}
        performanceMonitor={monitor}
      />
    </div>
  );
}
